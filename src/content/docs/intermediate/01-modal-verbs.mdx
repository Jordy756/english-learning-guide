## üìê Estructura: Afirmativo, Negativo, Pregunta

```typescript
// Los verbos modales NO cambian con la persona
// I/you/he/she/it/we/they + modal + verbo base

const buildSentence = (subject: string, modal: string, verb: string) => {
  return {
    afirmativo: `${subject} ${modal} ${verb}`,
    negativo: `${subject} ${modal} not ${verb}`,
    pregunta: `${modal} ${subject} ${verb}?`
  };
};

const example = buildSentence("She", "should", "work today");

console.log(example.afirmativo);  // "She should work today"
console.log(example.negativo);    // "She should not work today"
console.log(example.pregunta);    // "Should she work today?"
```
